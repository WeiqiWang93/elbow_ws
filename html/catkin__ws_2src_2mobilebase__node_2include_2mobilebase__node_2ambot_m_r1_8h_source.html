87  //
88  // Macros for BIG Endian machines. The MR1's ARM processor is little endian and we don't do conversions to network
89  // byte order (Big Endian) to limit the overhead on the MR1. You must swap bytes on all data types in the
90  // message structures.
91  //
92 #ifdef __BIG_ENDIAN__
93 
94 # define swap16(num) \
95  ((num) >> 8 | ((num) << 8))
96 
97 # define swap32(num) \
98  ((((num) >> 24) & 0xff) | \
99  (((num) << 8) &0xff0000) | \
100  (((num) >> 8) & 0xff00) | \
101  (((num) << 24) & 0xff000000))
102 
103 # define put16(num) swap16(num)
104 # define put32(num) swap32(num)
105 # define get16(num) swap16(num)
106 # define get32(num) swap32(num)
107 
108 #else
109 
110 # define put16(num) num
111 # define put32(num) num
112 # define get16(num) num
113 # define get32(num) num
114 
115 #endif
