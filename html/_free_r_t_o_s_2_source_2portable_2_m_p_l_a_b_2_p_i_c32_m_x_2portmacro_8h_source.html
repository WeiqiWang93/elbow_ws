79 /* Critical section management. */
80 #define portIPL_SHIFT ( 10UL )
81 #define portALL_IPL_BITS ( 0x3fUL << portIPL_SHIFT )
82 #define portSW0_BIT ( 0x01 << 8 )

84 /* This clears the IPL bits, then sets them to
85 configMAX_SYSCALL_INTERRUPT_PRIORITY. An extra check is performed if
86 configASSERT() is defined to ensure an assertion handler does not inadvertently
87 attempt to lower the IPL when the call to assert was triggered because the IPL
88 value was found to be above configMAX_SYSCALL_INTERRUPT_PRIORITY when an ISR
89 safe FreeRTOS API function was executed. ISR safe FreeRTOS API functions are
90 those that end in FromISR. FreeRTOS maintains a separate interrupt API to
91 ensure API function and interrupt entry is as fast and as simple as possible. */
92 #ifdef configASSERT
93  #define portDISABLE_INTERRUPTS() \
94  { \
95  uint32_t ulStatus; \
96  \
97  /* Mask interrupts at and below the kernel interrupt priority. */ \
98  ulStatus = _CP0_GET_STATUS(); \
99  \
100  /* Is the current IPL below configMAX_SYSCALL_INTERRUPT_PRIORITY? */ \
101  if( ( ( ulStatus & portALL_IPL_BITS ) >> portIPL_SHIFT ) < configMAX_SYSCALL_INTERRUPT_PRIORITY ) \
102  { \
103  ulStatus &= ~portALL_IPL_BITS; \
104  _CP0_SET_STATUS( ( ulStatus | ( configMAX_SYSCALL_INTERRUPT_PRIORITY << portIPL_SHIFT ) ) ); \
105  } \
106  }
107 #else /* configASSERT */
108  #define portDISABLE_INTERRUPTS() \
109  { \
110  uint32_t ulStatus; \
111  \
112  /* Mask interrupts at and below the kernel interrupt priority. */ \
113  ulStatus = _CP0_GET_STATUS(); \
114  ulStatus &= ~portALL_IPL_BITS; \
115  _CP0_SET_STATUS( ( ulStatus | ( configMAX_SYSCALL_INTERRUPT_PRIORITY << portIPL_SHIFT ) ) ); \
116  }
117 #endif /* configASSERT */
118 
119 #define portENABLE_INTERRUPTS() \
120 { \
121 uint32_t ulStatus; \
122  \
123  /* Unmask all interrupts. */ \
124  ulStatus = _CP0_GET_STATUS(); \
125  ulStatus &= ~portALL_IPL_BITS; \
126  _CP0_SET_STATUS( ulStatus ); \
127 }
