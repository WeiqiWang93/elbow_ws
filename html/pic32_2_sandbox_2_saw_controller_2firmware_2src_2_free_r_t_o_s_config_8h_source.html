146  /* portCONFIGURE_TIMER_FOR_RUN_TIME_STATS() is defined to call the
147  function that sets up timer 2 (T2) as the time base for the run time
148  statistics. T2 is a 16-bit timer and generates an interrupt when it
149  overflows.
150 
151  portALT_GET_RUN_TIME_COUNTER_VALUE() is defined to set its parameter to
152  the current run time stats counter/time value. The returned timer value
153  is 32-bits long, and is formed by shifting the T2 overflow count into the
154  top two bytes of a 32-bit number, then bitwise ORing the result with the
155  current T2 count value. */
156  void vSetupTimerForRunTimeStats( void );
157  unsigned long ulGetRunTimeStatsCounterValue( void );
158  #define portCONFIGURE_TIMER_FOR_RUN_TIME_STATS() vSetupTimerForRunTimeStats()
159 
160  /* T2 is 16-bits and generate an interrupt when it overflows. The
161  number of overflows is stored in ulTMR2OverflowCount. This macro
162  combines the current 16-bit counter value with the number of overflows
163  to return a 32-bit timer value. */
164  #define portALT_GET_RUN_TIME_COUNTER_VALUE( ulCountValue ) \
165  { \
166  extern volatile unsigned long ulTMR2OverflowCount; \
167  \
168  /* Disconnect the clock from the counter so it does not change \
169  while its value is being used. */ \
170  T4CONCLR = T4_ON; \
171  \
172  /* The number of overflows is shifted into the most significant \
173  two bytes of the returned 32-bit value. */ \
174  ulCountValue = ( ulTMR2OverflowCount << 16UL ); \
175  \
176  /* The current T2 counter value is used as the least \
177  significant two bytes of the returned 32-bit value. */ \
178  ulCountValue |= ( unsigned long ) ReadTimer4(); \
179  \
180  /* Reconnect the clock to the counter. */ \
181  T4CONSET = T4_ON; \
182  }
183 #endif /* __LANGUAGE_C__ */
