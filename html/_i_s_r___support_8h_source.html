34 /******************************************************************/
35 .macro portSAVE_CONTEXT
36 
37  /* Make room for the context. First save the current status so it can be
38  manipulated, and the cause and EPC registers so their original values are
39  captured. */
40  mfc0 k0, _CP0_CAUSE
41  addiu sp, sp, -portCONTEXT_SIZE
42  mfc0 k1, _CP0_STATUS
43 
44  /* Also save s6 and s5 so they can be used. Any nesting interrupts should
45  maintain the values of these registers across the ISR. */
46  sw s6, 44(sp)
47  sw s5, 40(sp)
48  sw k1, portSTATUS_STACK_LOCATION(sp)
49 
50  /* Prepare to enable interrupts above the current priority. */
51  srl k0, k0, 0xa
52  ins k1, k0, 10, 6
53  ins k1, zero, 1, 4
54 
55  /* s5 is used as the frame pointer. */
56  add s5, zero, sp
57 
58  /* Check the nesting count value. */
59  la k0, uxInterruptNesting
60  lw s6, (k0)
61 
62  /* If the nesting count is 0 then swap to the the system stack, otherwise
63  the system stack is already being used. */
64  bne s6, zero, 1f
65  nop
66 
67  /* Swap to the system stack. */
68  la sp, xISRStackTop
69  lw sp, (sp)
70 
71  /* Increment and save the nesting count. */
72 1: addiu s6, s6, 1
73  sw s6, 0(k0)
74 
75  /* s6 holds the EPC value, this is saved after interrupts are re-enabled. */
76  mfc0 s6, _CP0_EPC
77 
78  /* Re-enable interrupts. */
79  mtc0 k1, _CP0_STATUS
80 
81  /* Save the context into the space just created. s6 is saved again
82  here as it now contains the EPC value. No other s registers need be
83  saved. */
84  sw ra, 120(s5)
85  sw s8, 116(s5)
86  sw t9, 112(s5)
87  sw t8, 108(s5)
88  sw t7, 104(s5)
89  sw t6, 100(s5)
90  sw t5, 96(s5)
91  sw t4, 92(s5)
92  sw t3, 88(s5)
93  sw t2, 84(s5)
94  sw t1, 80(s5)
95  sw t0, 76(s5)
96  sw a3, 72(s5)
97  sw a2, 68(s5)
98  sw a1, 64(s5)
99  sw a0, 60(s5)
100  sw v1, 56(s5)
101  sw v0, 52(s5)
102  sw s6, portEPC_STACK_LOCATION(s5)
103  sw $1, 16(s5)
104 
105  /* s6 is used as a scratch register. */
106  mfhi s6
107  sw s6, 12(s5)
108  mflo s6
109  sw s6, 8(s5)
110 
111  /* Update the task stack pointer value if nesting is zero. */
112  la s6, uxInterruptNesting
113  lw s6, (s6)
114  addiu s6, s6, -1
115  bne s6, zero, 1f
116  nop
117 
118  /* Save the stack pointer. */
119  la s6, uxSavedTaskStackPointer
120  sw s5, (s6)
121 1:
122  .endm

124 /******************************************************************/
125 .macro portRESTORE_CONTEXT
126 
127  /* Restore the stack pointer from the TCB. This is only done if the
128  nesting count is 1. */
129  la s6, uxInterruptNesting
130  lw s6, (s6)
131  addiu s6, s6, -1
132  bne s6, zero, 1f
133  nop
134  la s6, uxSavedTaskStackPointer
135  lw s5, (s6)
136 
137  /* Restore the context. */
138 1: lw s6, 8(s5)
139  mtlo s6
140  lw s6, 12(s5)
141  mthi s6
142  lw $1, 16(s5)
143  /* s6 is loaded as it was used as a scratch register and therefore saved
144  as part of the interrupt context. */
145  lw s6, 44(s5)
146  lw v0, 52(s5)
147  lw v1, 56(s5)
148  lw a0, 60(s5)
149  lw a1, 64(s5)
150  lw a2, 68(s5)
151  lw a3, 72(s5)
152  lw t0, 76(s5)
153  lw t1, 80(s5)
154  lw t2, 84(s5)
155  lw t3, 88(s5)
156  lw t4, 92(s5)
157  lw t5, 96(s5)
158  lw t6, 100(s5)
159  lw t7, 104(s5)
160  lw t8, 108(s5)
161  lw t9, 112(s5)
162  lw s8, 116(s5)
163  lw ra, 120(s5)
164 
165  /* Protect access to the k registers, and others. */
166  di
167  ehb
168 
169  /* Decrement the nesting count. */
170  la k0, uxInterruptNesting
171  lw k1, (k0)
172  addiu k1, k1, -1
173  sw k1, 0(k0)
174 
175  lw k0, portSTATUS_STACK_LOCATION(s5)
176  lw k1, portEPC_STACK_LOCATION(s5)
177 
178  /* Leave the stack in its original state. First load sp from s5, then
179  restore s5 from the stack. */
180  add sp, zero, s5
181  lw s5, 40(sp)
182  addiu sp, sp, portCONTEXT_SIZE
183 
184  mtc0 k0, _CP0_STATUS
185  mtc0 k1, _CP0_EPC
186  ehb
187  eret
188  nop
189 
190  .endm
